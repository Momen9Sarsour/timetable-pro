<script>
    document.addEventListener('DOMContentLoaded', function () {
        document.querySelectorAll('.event-instructor, .event-room').forEach(field => {
            field.addEventListener('click', function () {
                const fieldData = this.dataset.field;
                const geneId = this.dataset.geneId;
                const currentValue = this.textContent.trim();

                const select = document.createElement('select');
                select.className = 'form-select';
                select.style.fontSize = '0.75rem';
                select.style.padding = '0.1rem';
                select.dataset.field = fieldData;
                select.dataset.geneId = geneId;

                // ŸÖÿ´ÿßŸÑ: ÿ¨ŸÑÿ® ÿßŸÑÿÆŸäÿßÿ±ÿßÿ™ (ÿ≥Ÿäÿ™ŸÖ ÿßÿ≥ÿ™ÿ®ÿØÿßŸÑŸá ÿ®ŸÄ AJAX ŸÑÿßÿ≠ŸÇÿßŸã)
                const options = fieldData === 'instructor'
                    ? @json($block->section->planSubject->subject->instructors->pluck('user.name', 'id'))
                    : @json($block->room->roomType->rooms->pluck('room_no', 'id'));

                Object.entries(options).forEach(([id, name]) => {
                    const option = document.createElement('option');
                    option.value = id;
                    option.textContent = name;
                    if (name === currentValue) option.selected = true;
                    select.appendChild(option);
                });

                this.replaceWith(select);
                $(select).select2({ width: '100%', dropdownAutoWidth: true });
                $(select).focus();

                $(select).on('select2:close', function () {
                    const selectedText = $(this).find(':selected').text();
                    const div = document.createElement('div');
                    div.className = this.dataset.field === 'instructor' ? 'event-instructor' : 'event-room';
                    div.dataset.field = this.dataset.field;
                    div.dataset.geneId = this.dataset.geneId;
                    div.textContent = selectedText;
                    $(this).replaceWith(div);

                    div.addEventListener('click', arguments.callee);
                });
            });
        });

        document.getElementById('exportPdfBtn').addEventListener('click', function () {
            alert('Export to PDF feature is under development.');
        });
    });
</script>
